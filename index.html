// Enhanced mobile touch handling let touchStartY = 0; let touchEndY = 0; const swipeThreshold = 50; document.addEventListener('touchstart', function(e) { touchStartY = e.changedTouches[0].screenY; }, { passive: true }); document.addEventListener('touchend', function(e) { touchEndY = e.changedTouches[0].screenY; handleVerticalSwipe(); }, { passive: true }); function handleVerticalSwipe() { const diff = touchStartY - touchEndY; if (Math.abs(diff) > swipeThreshold) { if (diff > 0) { // Swipe up - next slide goToSlide(currentSlide + 1); } else { // Swipe down - previous slide goToSlide(currentSlide - 1); } } } // Prevent zoom on double-tap document.addEventListener('dblclick', function(e) { e.preventDefault(); }, { passive: false });
